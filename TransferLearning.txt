# ===== Import Libraries =====
import tensorflow as tf
from tensorflow.keras import layers, models
from tensorflow.keras.preprocessing.image import ImageDataGenerator
from tensorflow.keras.applications import MobileNetV2
import matplotlib.pyplot as plt

# ===== Step 1: Load and Preprocess Data =====
train_gen = ImageDataGenerator(rescale=1./255)
val_gen = ImageDataGenerator(rescale=1./255)

train_data = train_gen.flow_from_directory(
    'data',           # Folder path for training data
    target_size=(224, 224),
    batch_size=16,
    class_mode='categorical'
)

val_data = val_gen.flow_from_directory(
    'data',             # Folder path for validation data
    target_size=(224, 224),
    batch_size=16,
    class_mode='categorical'
)

# ===== Step 2: Load Pretrained VGG16 Model =====
base_model = tf.keras.applications.MobileNetV2(
    input_shape=(224, 224, 3),
    include_top=False,      # remove default classifier layers
    weights='imagenet'
)

# Freeze convolutional layers
for layer in base_model.layers:
    layer.trainable = False

# ===== Step 3: Add Custom Classifier =====
model = models.Sequential([
    base_model,
    layers.Flatten(),
    layers.Dense(256, activation='relu'),
    layers.Dropout(0.3),
    layers.Dense(train_data.num_classes, activation='softmax')
])

# ===== Step 4: Compile the Model =====
model.compile(
    optimizer='adam',
    loss='categorical_crossentropy',
    metrics=['accuracy']
)

# ===== Step 5: Train the Model =====
history = model.fit(
    train_data,
    validation_data=val_data,
    epochs=3
)

# ===== Step 6: Evaluate Model =====
loss, acc = model.evaluate(val_data)
print(f"✅ Validation Accuracy: {acc*100:.2f}%")

# ===== Step 7: Save Model =====
model.save("vgg16_transfer_learning.h5")
print("✅ Model saved as vgg16_transfer_learning.h5")

# ===== Step 8: Plot Training and Validation Graphs =====
plt.figure(figsize=(12,5))

# Accuracy Graph
plt.subplot(1,2,1)
plt.plot(history.history['accuracy'], label='Train Accuracy')
plt.plot(history.history['val_accuracy'], label='Validation Accuracy')
plt.title('Model Accuracy')
plt.xlabel('Epochs')
plt.ylabel('Accuracy')
plt.legend()
plt.grid(True)

# Loss Graph
plt.subplot(1,2,2)
plt.plot(history.history['loss'], label='Train Loss')
plt.plot(history.history['val_loss'], label='Validation Loss')
plt.title('Model Loss')
plt.xlabel('Epochs')
plt.ylabel('Loss')
plt.legend()
plt.grid(True)

plt.show()
